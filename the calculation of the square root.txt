(define (qub guess x)
  (if (isOk? guess x)
      guess
      (qub (improve guess x)
           x)))
(define (improve guess x)
  (/ (+ (/ x (square guess))
        (* 2 guess))
     3))
(define (square x)
  (* x x))
(define (isOk? guess x)
  (< (abs (- guess x)) (dividend guess)))
(define (dividend guess)
  (/ guess 100))
